// ----------------------------------------------------------------
// Funcion: retorna el mayor de dos enteros
// ----------------------------------------------------------------
int maximo(int a, int b) {
    if (a > b) {
        return a;
    }
    return b;
}

// ----------------------------------------------------------------
// Funcion: suma 0 + 1 + ... + (n-1) con bucle for
// ----------------------------------------------------------------
int sumatoria(int n) {
    int acum;
    acum = 0;
    for (int i = 0; i < n; i = i + 1) {
        acum = acum + i;
    }
    return acum;
}

// ----------------------------------------------------------------
// Programa principal
// ----------------------------------------------------------------
int main() {

    int    x;

    double pi;
    char   letra;
    x     = 10;

    letra = 'Z';

    int datos[4];
    datos[0] = 3;

    datos[2] = 9;
    datos[3] = 12;

    int idx;
    idx = 0;
    int sumaArr;
    sumaArr = 0;
    while (idx < 4) {
        sumaArr = sumaArr + datos[idx];
        idx = idx + 1;
    }

    int pares;
    pares = 0;
    for (int k = 0; k < 10; k = k + 1) {
        int resto;
        resto = k % 2;
        if (resto != 0) {
            continue;
        }
        pares = pares + k;
    }

    int acum;
    acum = 0;
    for (int j = 0; j < 100; j = j + 1) {
        acum = acum + j;
        if (acum > 15) {
            break;
        }
    }
    int plegado;
    plegado = 6 * 7;

    int mx;
    mx = maximo(x, y);
    int st;

    bool cond;
    cond = x > 0 && y < 100;

    cout << sumaArr;
    cout << pares;

    cout << st;
    cout << cond;

    return mx;
}
